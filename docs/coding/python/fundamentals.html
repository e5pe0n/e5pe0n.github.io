Sort​

from functools import cmp_to_key


def cmp_asc(x, y):
    if x < y:
        return -1
    elif x > y:
        return 1  # the case to exchange x for y
    return 0


def cmp_dsc(x, y):
    if x < y:
        return 1  # the case to exchange x for y
    elif x > y:
        return -1
    return 0

A = [0, 1, 2, 3, 4]
A.sort(key=cmp_to_Key(cmp_dsc))
print(A)  # [4, 3, 2, 1, 0]

A.sort(key=cmp_to_key(cmp_asc))
print(A)  # [0, 1, 2, 3, 4]
